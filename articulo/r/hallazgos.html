# Creamos dos bases de datos para comparar:
# datos_1m: contiene los datos de los hijos que han fallecido.
# datos_0m: contienen los datos de los hijos de las madres que no han perdido
#           ningún hijo.
# Note que si alguna madre perdió un hijo antes del año 2000, y no ha perdido
# hijos posteriormente, quedará dentro de la base datos_0m

si_ha_perdido_hijos12 <-
  datos_orig[edad_muerte_meses <= 12,
             .(hijos_vivos = sum(hijo_vivo_muerto) != mean(nro_hijos)),
             by = id_madre][hijos_vivos == TRUE, id_madre]

# datos_1m <- datos_orig[edad_muerte_meses <= 12 & !is.na(edad_muerte_meses)]
datos_1m <- datos_orig[id_madre %in% si_ha_perdido_hijos12]

no_ha_perdido_hijos12 <-
  datos_orig[,
             .(hijos_vivos = sum(hijo_vivo_muerto) == mean(nro_hijos)),
